import gym
import numpy as np
import random
from tqdm import tqdm

env = gym.make('CartPole-v1')
num_state = env.observation_space.shape[0]
num_action = env.action_space.n
num_episode = 5000
memory = []

# Hyper parameter
epsilon = 0.3
gamma = 0.9
batch_size = 32

# DQN Modeling
from tensorflow.keras import Sequential
from tensorflow.keras.layers import Dense

model = Sequential()
model.add(Dense(24, input_dim= num_state, activation='relu'))
model.add(Dense(24, activation='relu'))
model.add(Dense(num_action, activation='linear'))
model.compile(loss='mse', optimizer='adam')

# DQN Learning
for episode in tqdm(range(num_episode)):
    state = env.reset()
    state = np.array(state).reshape(1, num_state)
    done = False
    while not done:
        if np.random.uniform() < epsilon:
            action = env.action_space.sample()
        else:
            q_value = model.predict(state)
            action = np.argmax(q_value[0])
        next_state, reward, done, info = env.step(action)
        next_state = np.array(next_state).reshape(1, num_state)
        # Memory
        memory.append((state, action, reward, next_state, done))
        
        state = next_state
    
    # Replay
    if len(memory) > batch_size:
        mini_batch = random.sample(memory, batch_size)
        for state, action, reward, next_state, done in mini_batch:
            target = reward
            if not done:
                target = reward + gamma * (np.max(model.predict(next_state)[0]))
            q_value = model.predict(state)
            q_value[0][action] = target
            model.fit(state, q_value, epochs=1, verbose=0)

env.close()
